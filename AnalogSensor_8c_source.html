107void ProcessADCData(uint16_t* adc1_data, uint16_t* adc2_data, uint16_t* adc3_data) {
108 ADCSample sample = {0}; // Initialize all channels to 0
109
110 // Process ADC1 data (PA0-PA5: channels 0-5)
111 for (int i = 0; i < 6; i++) {
112 sample.adc[i] = adc1_data[i];
113 }
114
115 // Process ADC2 data (PA6-PA7: channels 6-7, PB0-PB1: channels 8-9, PC4-PC5: channels 14-15)
116 sample.adc[6] = adc2_data[0]; // PA6
117 sample.adc[7] = adc2_data[1]; // PA7
118 sample.adc[8] = adc2_data[2]; // PB0
119 sample.adc[9] = adc2_data[3]; // PB1
120 sample.adc[14] = adc2_data[4]; // PC4
121 sample.adc[15] = adc2_data[5]; // PC5
122
123 // Process ADC3 data (PC0-PC3: channels 10-13)
124 for (int i = 0; i < 4; i++) {
125 sample.adc[i + 10] = adc3_data[i];
126 }
127
128 addSampleToBuffer(sample);
129
130 // Optional: UART debug output (adjust as needed)
131 char uart_buf[100];
132 snprintf(uart_buf, sizeof(uart_buf), "ADC0: %4d, ADC7: %4d, ADC10: %4d\r\n",
133 sample.adc[0], sample.adc[7], sample.adc[10]);
134 HAL_UART_Transmit(&huart2, (uint8_t*)uart_buf, strlen(uart_buf), HAL_MAX_DELAY);
